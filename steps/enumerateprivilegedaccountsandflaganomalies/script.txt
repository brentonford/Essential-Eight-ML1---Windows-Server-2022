# ISM-0445: Checks for accounts that appear to violate the two-account model.
# Flags: built-in admin enabled, accounts with no password age (never changed),
# accounts with very long password ages, accounts that appear to be standard
# user accounts elevated to admin (no .admin / adm- naming convention).

$adminMembers = Get-LocalGroupMember -Group "Administrators" -ErrorAction Stop

Write-Output "Administrators group members: $($adminMembers.Count)"
Write-Output ""

$flags = @()

foreach ($member in $adminMembers) {
    if ($member.PrincipalSource -ne 'Local' -or $member.ObjectClass -ne 'User') {
        Write-Output "  $($member.Name) -- non-local ($($member.PrincipalSource) $($member.ObjectClass)) -- cannot audit locally"
        continue
    }

    $account = Get-LocalUser -SID $member.SID -ErrorAction SilentlyContinue

    if ($null -eq $account) {
        Write-Output "  $($member.Name) -- local user lookup failed"
        continue
    }

    $passwordAge = if ($account.PasswordLastSet) { ((Get-Date) - $account.PasswordLastSet).Days } else { 9999 }
    $isBuiltIn   = $account.SID.Value.EndsWith("-500")
    $lastLogon   = if ($account.LastLogon)       { $account.LastLogon.ToString("yyyy-MM-dd") }   else { "Never" }

    $accountFlags = @()
    if ($isBuiltIn -and $account.Enabled) { $accountFlags += "BUILT-IN ENABLED" }
    if ($passwordAge -gt 365)             { $accountFlags += "PASSWORD AGE ${passwordAge}d > 365d" }
    if ($passwordAge -eq 9999)            { $accountFlags += "PASSWORD NEVER SET" }

    $status = if ($accountFlags.Count -gt 0) { "FLAG" } else { "OK" }

    Write-Output "  [$status] $($account.Name)"
    Write-Output "    Enabled      : $($account.Enabled)"
    Write-Output "    Built-in     : $isBuiltIn"
    Write-Output "    Password age : $passwordAge days"
    Write-Output "    Last logon   : $lastLogon"

    if ($accountFlags.Count -gt 0) {
        $accountFlags | ForEach-Object { Write-Output "    *** FLAG: $_" }
        $flags += "$($account.Name): $($accountFlags -join ', ')"
    }
    Write-Output ""
}

Write-Output "Total flags: $($flags.Count)"
