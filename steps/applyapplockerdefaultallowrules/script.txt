$allowPathsExeRaw    = "{allowPathsExe}"
$allowPathsScriptRaw = "{allowPathsScript}"

class AppLockerAllowRuleXml {
    [string]$path
    [string]$sid

    AppLockerAllowRuleXml([string]$path, [string]$sid) {
        $this.path = $path
        $this.sid = $sid
    }

    [string] ToXml() {
        $ruleId = [guid]::NewGuid().Guid
        $escapedPath = [System.Security.SecurityElement]::Escape($this.path)
        $sidLabel = if ($this.sid -eq 'S-1-1-0') { 'Everyone' } else { $this.sid }
        $escapedName = [System.Security.SecurityElement]::Escape("Allow $sidLabel - $($this.path)")
        $escapedDesc = [System.Security.SecurityElement]::Escape("Allows execution from $($this.path)")

        return @"
    <FilePathRule Id="$ruleId" Name="$escapedName" Description="$escapedDesc" UserOrGroupSid="$($this.sid)" Action="Allow">
      <Conditions>
        <FilePathCondition Path="$escapedPath"/>
      </Conditions>
    </FilePathRule>
"@
    }
}

function Parse-CsvParameter {
    param([string]$raw)

    $result = @()
    foreach ($entry in ($raw -split ",")) {
        $trimmed = $entry.Trim()
        if ($trimmed -eq "") {
            continue
        }
        $result += $trimmed
    }
    return $result
}

$allowPathsExe    = Parse-CsvParameter -raw $allowPathsExeRaw
$allowPathsScript = Parse-CsvParameter -raw $allowPathsScriptRaw

if ($allowPathsExe.Count -eq 0) {
    Write-Error "Parameter allowPathsExe is empty. At least one Exe allow path is required to avoid a default-deny lockout."
    exit 1
}
if ($allowPathsScript.Count -eq 0) {
    Write-Error "Parameter allowPathsScript is empty. At least one Script allow path is required."
    exit 1
}

Write-Output "Exe allow paths    : $($allowPathsExe.Count) + 1 hardcoded admin rule"
Write-Output "Script allow paths : $($allowPathsScript.Count) + 1 hardcoded admin rule"

# Hardcoded safety: Administrators-all-files in BOTH collections (always added).
$exeRulesXml = @()
foreach ($path in $allowPathsExe) {
    $exeRulesXml += [AppLockerAllowRuleXml]::new($path, 'S-1-1-0').ToXml()
}
$exeRulesXml += [AppLockerAllowRuleXml]::new('*', 'S-1-5-32-544').ToXml()

$scriptRulesXml = @()
foreach ($path in $allowPathsScript) {
    $scriptRulesXml += [AppLockerAllowRuleXml]::new($path, 'S-1-1-0').ToXml()
}
$scriptRulesXml += [AppLockerAllowRuleXml]::new('*', 'S-1-5-32-544').ToXml()

$policyXml = @"
<AppLockerPolicy Version="1">
  <RuleCollection Type="Exe" EnforcementMode="AuditOnly">
$($exeRulesXml -join "`n")
  </RuleCollection>
  <RuleCollection Type="Script" EnforcementMode="AuditOnly">
$($scriptRulesXml -join "`n")
  </RuleCollection>
</AppLockerPolicy>
"@

$tempPolicyPath = "$env:TEMP\applocker_allow.xml"
$policyXml | Out-File -FilePath $tempPolicyPath -Encoding UTF8

Set-AppLockerPolicy -XmlPolicy $tempPolicyPath -Merge
Write-Output "Allow rules applied successfully (with hardcoded Administrators-all-files safety rule in both collections)."

Remove-Item -Path $tempPolicyPath -Force
