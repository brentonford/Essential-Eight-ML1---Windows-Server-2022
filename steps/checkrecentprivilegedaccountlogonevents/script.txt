# Surfaces recent privileged account logon events from the Security event log.
# Event 4672: Special privileges assigned to new logon (admin logon indicator)
# Event 4624: Successful logon
# Event 4625: Failed logon
# Lookback: 7 days

$cutoff = (Get-Date).AddDays(-7)
$events = @()

$eventIds = @(4672, 4625)

foreach ($id in $eventIds) {
    $found = Get-WinEvent -FilterHashtable @{
        LogName   = "Security"
        Id        = $id
        StartTime = $cutoff
    } -ErrorAction SilentlyContinue

    if ($found) { $events += $found }
}

$adminLogons  = ($events | Where-Object { $_.Id -eq 4672 }).Count
$failedLogons = ($events | Where-Object { $_.Id -eq 4625 }).Count

Write-Output "Privileged logon events (last 7 days):"
Write-Output "  Privileged logons (4672) : $adminLogons"
Write-Output "  Failed logons    (4625)  : $failedLogons"

if ($failedLogons -gt 10) {
    Write-Output ""
    Write-Output "WARN: $failedLogons failed logon attempts in the last 7 days."
    Write-Output "Review Security event log for potential brute-force activity."
}
