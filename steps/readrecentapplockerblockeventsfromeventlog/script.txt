# AppLocker block events:
# Event ID 8004 = EXE or DLL blocked
# Event ID 8007 = Script blocked
# Event ID 8022 = MSI/package blocked
# This Step is informational only -- always exits 0 regardless of event count.
$appLockerLogNames = @(
    "Microsoft-Windows-AppLocker/EXE and DLL",
    "Microsoft-Windows-AppLocker/Script"
)

$blockEventIds = @(8004, 8007, 8022)
$lookbackHours = 168  # 7 days
$cutoffTime = (Get-Date).AddHours(-$lookbackHours)

$blockEvents = @()

foreach ($logName in $appLockerLogNames) {
    $events = Get-WinEvent -LogName $logName -ErrorAction SilentlyContinue |
        Where-Object { $_.Id -in $blockEventIds -and $_.TimeCreated -gt $cutoffTime }

    if ($null -ne $events) {
        $blockEvents += $events
    }
}

Write-Output "AppLocker block events in the last 7 days: $($blockEvents.Count)"

if ($blockEvents.Count -eq 0) {
    Write-Output "No block events recorded. Policy may be newly applied or no violations have occurred."
} else {
    Write-Output "Block events recorded."
}

$blockEvents |
    Sort-Object TimeCreated -Descending |
    Select-Object -First 10 |
    ForEach-Object {
        $firstLine = ($_.Message -split "`n")[0].Trim()
        Write-Output "  [$($_.TimeCreated)] ID $($_.Id) -- $firstLine"
    }
