$expectedEnforcementMode = "{expectedEnforcementMode}".Trim()

$validModes = @('AuditOnly', 'Enabled')
if ($expectedEnforcementMode -notin $validModes) {
    Write-Error "Parameter expectedEnforcementMode='$expectedEnforcementMode' is invalid. Must be one of: $($validModes -join ', ')"
    exit 1
}

$effectivePolicy = Get-AppLockerPolicy -Effective -ErrorAction SilentlyContinue

if ($null -eq $effectivePolicy) {
    Write-Output "CHECK Enforcement Mode : FAIL -- no AppLocker policy found"
    exit 1
}

$ruleCount = ($effectivePolicy.RuleCollections | ForEach-Object { $_.Count } | Measure-Object -Sum).Sum
if ($null -eq $ruleCount) {
    $ruleCount = 0
}

Write-Output "Total rules in effective policy: $ruleCount"
Write-Output "Expected enforcement mode      : $expectedEnforcementMode"

if ($ruleCount -eq 0) {
    Write-Output "CHECK Enforcement Mode : FAIL -- effective policy contains zero rules"
    exit 1
}

$nonCompliantCollections = @()

foreach ($collection in $effectivePolicy.RuleCollections) {
    $mode = $collection.EnforcementMode
    Write-Output "Rule collection '$($collection.RuleCollectionType)' enforcement: $mode"

    if ($mode -ne $expectedEnforcementMode) {
        $nonCompliantCollections += $collection.RuleCollectionType
    }
}

if ($nonCompliantCollections.Count -gt 0) {
    Write-Output "CHECK Enforcement Mode : FAIL -- collections not in '$expectedEnforcementMode' mode: $($nonCompliantCollections -join ', ')"
    exit 1
}

Write-Output "CHECK Enforcement Mode : PASS"
