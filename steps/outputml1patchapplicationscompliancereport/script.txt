$hostname   = $env:COMPUTERNAME
$reportDate = Get-Date -Format "yyyy-MM-dd HH:mm:ss"

# --- PSWindowsUpdate check ---
$pswuModule = Get-Module -ListAvailable -Name PSWindowsUpdate
$pswuCheck = if ($null -ne $pswuModule) { "PASS" } else { "FAIL" }

# --- Outstanding updates ---
$outstandingUpdates = @()
$outstandingCheck = "SKIP"
if ($null -ne $pswuModule) {
    Import-Module PSWindowsUpdate
    $outstandingUpdates = Get-WindowsUpdate -Category "Application" -NotTitle "Upgrade to" -ErrorAction SilentlyContinue
    $outstandingCheck = if ($null -eq $outstandingUpdates -or $outstandingUpdates.Count -eq 0) { "PASS" } else { "FAIL" }
}

# --- Update recency ---
$daysSinceUpdate = -1
$recencyCheck = "SKIP"
if ($null -ne $pswuModule) {
    $recentHistory = Get-WUHistory -MaxDate (Get-Date) -ErrorAction SilentlyContinue |
        Where-Object { $_.Categories -match "Application" } |
        Sort-Object Date -Descending |
        Select-Object -First 1

    if ($null -ne $recentHistory) {
        $daysSinceUpdate = ((Get-Date) - $recentHistory.Date).Days
        $recencyCheck = if ($daysSinceUpdate -le 30) { "PASS" } else { "FAIL" }
    } else {
        $recencyCheck = "WARN"
    }
}

# --- EOL scan ---
$eolApplications = @(
    "Adobe Flash Player", "Internet Explorer", "Java 6", "Java 7",
    "Java(TM) 6", "Java(TM) 7", "Microsoft Silverlight",
    "Microsoft Visual C++ 2005", "Microsoft Visual C++ 2008",
    "Python 2", "7-Zip 1", "7-Zip 2"
)

$registryPaths = @(
    "HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall\*",
    "HKLM:\Software\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*"
)

$installedApps = $registryPaths | ForEach-Object {
    Get-ItemProperty -Path $_ -ErrorAction SilentlyContinue
} | Where-Object { $_.DisplayName }

$foundEol = @()
foreach ($eolEntry in $eolApplications) {
    $foundEol += $installedApps | Where-Object { $_.DisplayName -like "*$eolEntry*" }
}

$eolCheck = if ($foundEol.Count -eq 0) { "PASS" } else { "FAIL" }

# --- Reboot check ---
$rebootPending = (Test-Path "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Component Based Servicing\RebootPending") -or
                 (Test-Path "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\WindowsUpdate\Auto Update\RebootRequired")

# --- Overall ---
$checksWithResults = @($pswuCheck, $outstandingCheck, $recencyCheck, $eolCheck) |
    Where-Object { $_ -ne "SKIP" }

$overallStatus = if ($checksWithResults -contains "FAIL") { "NON-COMPLIANT" } else { "COMPLIANT" }

$separator = "=" * 60

Write-Output $separator
Write-Output "  ESSENTIAL EIGHT ML1 — PATCH APPLICATIONS"
Write-Output "  ASD Control: Mitigation Strategy 2"
Write-Output "  Reference  : cyber.gov.au/essential-eight"
Write-Output $separator
Write-Output "  Server     : $hostname"
Write-Output "  Report Date: $reportDate"
Write-Output "  Generated  : AttuneOps"
Write-Output $separator
Write-Output ""
Write-Output "  CHECKS"
Write-Output ""
Write-Output "  [1] PSWindowsUpdate Module"
Write-Output "      Result : $pswuCheck"
Write-Output ""
Write-Output "  [2] Outstanding Application Security Updates (Windows Update)"
Write-Output "      Pending: $($outstandingUpdates.Count)"
Write-Output "      Result : $outstandingCheck"
Write-Output ""
Write-Output "  [3] Application Update Recency"
Write-Output "      Days since last update : $(if ($daysSinceUpdate -ge 0) { $daysSinceUpdate } else { 'Unknown' })"
Write-Output "      ML1 threshold          : 30 days"
Write-Output "      Result                 : $recencyCheck"
Write-Output ""
Write-Output "  [4] EOL Applications"
Write-Output "      EOL apps found : $($foundEol.Count)"
if ($foundEol.Count -gt 0) {
    $foundEol | ForEach-Object {
        Write-Output "        - $($_.DisplayName) $($_.DisplayVersion)"
    }
}
Write-Output "      Result         : $eolCheck"
Write-Output ""
Write-Output "  [5] Reboot Pending"
Write-Output "      Status : $(if ($rebootPending) { 'YES — schedule maintenance window reboot' } else { 'No' })"
Write-Output ""
Write-Output $separator
Write-Output "  OVERALL STATUS : $overallStatus"
Write-Output $separator

if ($overallStatus -eq "NON-COMPLIANT") {
    Write-Output ""
    Write-Output "  REMEDIATION: Run the Configure Patch Applications - ML1"
    Write-Output "  Blueprint against this server to restore compliance."
    exit 1
}
