$svc = Get-Service -Name AppIDSvc -ErrorAction SilentlyContinue

if ($null -eq $svc) {
    Write-Output "AppIDSvc Status      : NOT FOUND"
    Write-Output "CHECK AppIDSvc       : FAIL -- service not present on this system"
    exit 1
}

$startType = (Get-CimInstance -ClassName Win32_Service -Filter "Name='AppIDSvc'").StartMode

Write-Output "AppIDSvc Status      : $($svc.Status)"
Write-Output "AppIDSvc StartType   : $startType"

if ($svc.Status -ne "Running" -or $startType -ne "Auto") {
    Write-Output "CHECK AppIDSvc       : FAIL"
    exit 1
}

Write-Output "CHECK AppIDSvc       : PASS"
