$archivePath = "{policyExportPath}"
$timestamp = Get-Date -Format "yyyyMMdd-HHmmss"
$hostname = $env:COMPUTERNAME
$archiveFile = Join-Path -Path $archivePath -ChildPath "applocker-baseline-$hostname-$timestamp.xml"

$effectivePolicy = Get-AppLockerPolicy -Effective
$effectivePolicy.ToXml() | Out-File -FilePath $archiveFile -Encoding UTF8

Write-Output "Baseline policy archived to: $archiveFile"
Write-Output "Timestamp : $timestamp"
Write-Output "Server    : $hostname"

$ruleCount = ($effectivePolicy.RuleCollections | ForEach-Object { $_.Count } | Measure-Object -Sum).Sum
Write-Output "Total rules archived: $ruleCount"
