$svc = Get-Service -Name AppIDSvc -ErrorAction SilentlyContinue

if ($null -eq $svc) {
    Write-Error "Application Identity service (AppIDSvc) was not found on this system."
    exit 1
}

$svcRegPath = "HKLM:\SYSTEM\CurrentControlSet\Services\AppIDSvc"
Set-ItemProperty -Path $svcRegPath -Name "Start" -Value 2 -Type DWord

# Verify the write actually took (Start values: 2=Automatic, 3=Manual, 4=Disabled).
$startValue = (Get-ItemProperty -Path $svcRegPath -Name "Start").Start
if ($startValue -ne 2) {
    Write-Error "Failed to set AppIDSvc Start value to 2 (Automatic). Current value: $startValue"
    exit 1
}
Write-Output "AppIDSvc startup type set to Automatic (registry Start=2, verified)."

if ($svc.Status -ne "Running") {
    Start-Service -Name AppIDSvc
    Start-Sleep -Seconds 3
}

$svc.Refresh()
Write-Output "AppIDSvc status : $($svc.Status)"

if ($svc.Status -ne "Running") {
    Write-Error "Application Identity service failed to start."
    exit 1
}

Write-Output "Application Identity service is running."
