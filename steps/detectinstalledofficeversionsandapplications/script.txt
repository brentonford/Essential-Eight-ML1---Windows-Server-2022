$officeVersions = @("16.0", "15.0", "14.0")
$officeApps     = @("Word", "Excel", "PowerPoint", "Access", "Publisher", "Outlook", "Visio", "Project")

$detected = @()

foreach ($version in $officeVersions) {
    foreach ($app in $officeApps) {
        $regPath = "HKLM:\SOFTWARE\Microsoft\Office\$version\$app"
        if (-not (Test-Path $regPath)) {
            continue
        }
        $detected += [PSCustomObject]@{
            Version = $version
            App     = $app
            RegBase = "HKLM:\SOFTWARE\Policies\Microsoft\Office\$version\$app\Security"
        }
        Write-Output "Detected: Office $version — $app"
    }
}

if ($detected.Count -eq 0) {
    Write-Output ""
    Write-Output "No Microsoft Office installation detected on this server."
    Write-Output "E3 STATUS: NOT APPLICABLE"
    Write-Output "No configuration required. Document this result in the compliance report."
    exit 0
}

Write-Output ""
Write-Output "Total Office app registrations detected: $($detected.Count)"
