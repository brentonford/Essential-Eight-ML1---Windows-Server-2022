$denyPathsExeRaw    = "{denyPathsExe}"
$denyPathsScriptRaw = "{denyPathsScript}"

function Parse-CsvParameter {
    param([string]$raw)

    $result = @()
    foreach ($entry in ($raw -split ",")) {
        $trimmed = $entry.Trim()
        if ($trimmed -eq "") {
            continue
        }
        $result += $trimmed
    }
    return $result
}

function Get-DenyPathsFromPolicyXml {
    param([xml]$policyXml, [string]$collectionType)

    $result = @()
    $xpath = "//RuleCollection[@Type='$collectionType']/FilePathRule[@Action='Deny']/Conditions/FilePathCondition"
    foreach ($node in $policyXml.SelectNodes($xpath)) {
        $result += $node.GetAttribute('Path')
    }
    return $result
}

$requiredExePaths    = Parse-CsvParameter -raw $denyPathsExeRaw
$requiredScriptPaths = Parse-CsvParameter -raw $denyPathsScriptRaw

if ($requiredExePaths.Count -eq 0 -and $requiredScriptPaths.Count -eq 0) {
    Write-Error "Both denyPathsExe and denyPathsScript are empty -- nothing to verify."
    exit 1
}

$effectivePolicy = Get-AppLockerPolicy -Effective -ErrorAction SilentlyContinue

if ($null -eq $effectivePolicy) {
    Write-Output "CHECK Block Rules : FAIL -- no AppLocker policy found"
    exit 1
}

# Parse policy as XML for reliable rule iteration.
$policyXml     = [xml]$effectivePolicy.ToXml()
$exePresent    = Get-DenyPathsFromPolicyXml -policyXml $policyXml -collectionType 'Exe'
$scriptPresent = Get-DenyPathsFromPolicyXml -policyXml $policyXml -collectionType 'Script'

Write-Output "Exe deny paths found in policy    : $($exePresent.Count)"
Write-Output "Script deny paths found in policy : $($scriptPresent.Count)"

$missingExe = @()
foreach ($required in $requiredExePaths) {
    if ($required -in $exePresent) {
        Write-Output "Exe    PRESENT : $required"
        continue
    }
    Write-Output "Exe    MISSING : $required"
    $missingExe += $required
}

$missingScript = @()
foreach ($required in $requiredScriptPaths) {
    if ($required -in $scriptPresent) {
        Write-Output "Script PRESENT : $required"
        continue
    }
    Write-Output "Script MISSING : $required"
    $missingScript += $required
}

$totalMissing = $missingExe.Count + $missingScript.Count

if ($totalMissing -gt 0) {
    Write-Output "CHECK Block Rules : FAIL -- $totalMissing required block rule(s) missing (Exe: $($missingExe.Count), Script: $($missingScript.Count))"
    exit 1
}

Write-Output "CHECK Block Rules : PASS -- all required paths present in both collections"
