$applyPthMitigation = "{applyPthMitigation}".Trim().ToLower()
$hostname           = $env:COMPUTERNAME
$reportDate         = Get-Date -Format "yyyy-MM-dd HH:mm:ss"

# Built-in admin check
$builtInAdmin   = Get-LocalUser | Where-Object { $_.SID.Value.EndsWith("-500") }
$builtInEnabled = $null -ne $builtInAdmin -and $builtInAdmin.Enabled
$ism0445        = if (-not $builtInEnabled) { "PASS" } else { "FAIL" }

# Admin count
$adminCount = (Get-LocalGroupMember -Group "Administrators" -ErrorAction Stop).Count

# Password policy
$seceditExport = "$env:TEMP\secpol_report.cfg"
secedit /export /cfg $seceditExport /quiet
$content       = Get-Content $seceditExport
$getValue      = { param($k) $l = $content | Where-Object { $_ -match "^$k\s*=" }; if ($l) { ($l -split '=')[1].Trim() } else { "0" } }
$minLength     = & $getValue "MinimumPasswordLength"
$complexity    = & $getValue "PasswordComplexity"
Remove-Item $seceditExport -Force -ErrorAction SilentlyContinue

$passCheck   = ([int]$minLength -ge 14 -and $complexity -eq "1")
$ism0445Pass = if ($ism0445 -eq "PASS" -and $passCheck) { "PASS" } else { "FAIL" }

# User Rights Assignment
$seceditExport2 = "$env:TEMP\secpol_ura_report.cfg"
secedit /export /cfg $seceditExport2 /quiet
$content2       = Get-Content $seceditExport2
$interactiveLogon = ($content2 | Where-Object { $_ -match "^SeInteractiveLogonRight" }) -replace '.*=\s*', ''
$denyNetwork      = ($content2 | Where-Object { $_ -match "^SeDenyNetworkLogonRight" })    -replace '.*=\s*', ''
Remove-Item $seceditExport2 -Force -ErrorAction SilentlyContinue

$ism1688 = if ($interactiveLogon -match 'S-1-5-32-544') { "PASS" } else { "FAIL" }

# ISM-1175 result depends on the declared posture (applyPthMitigation parameter)
# AND the observed state of SeDenyNetworkLogonRight on the host.
$s114Present = $denyNetwork -match 'S-1-5-114'

if ($applyPthMitigation -eq "true") {
    if ($s114Present) {
        $ism1175       = "PASS"
        $ism1175Detail = "Full local PtH mitigation applied."
    } else {
        $ism1175       = "FAIL"
        $ism1175Detail = "applyPthMitigation=true but S-1-5-114 missing from SeDenyNetworkLogonRight. Re-run Configure Step 5."
    }
} else {
    if ($s114Present) {
        $ism1175       = "WARN"
        $ism1175Detail = "applyPthMitigation=false but S-1-5-114 is present in the deny rule. WinRM management with local admin accounts is likely broken on this host."
    } else {
        $ism1175       = "PASS (partial)"
        $ism1175Detail = "Partial implementation by design. Compensating controls: ISM-1380 (SAW), network segmentation, audit logging."
    }
}

$separator = "=" * 60

Write-Output $separator
Write-Output "  ESSENTIAL EIGHT ML1 -- RESTRICT ADMINISTRATIVE PRIVILEGES"
Write-Output "  ASD Control: Mitigation Strategy 5"
Write-Output "  ISM Controls: 0445, 1175, 1380, 1507, 1688, 1689"
Write-Output "  Reference  : cyber.gov.au/essential-eight"
Write-Output $separator
Write-Output "  Server     : $hostname"
Write-Output "  Report Date: $reportDate"
Write-Output "  Generated  : AttuneOps"
Write-Output $separator
Write-Output ""
Write-Output "  CHECKS"
Write-Output ""
Write-Output "  [ISM-0445] Dedicated Privileged Accounts"
Write-Output "    Admin group members       : $adminCount"
Write-Output "    Built-in admin enabled    : $builtInEnabled"
Write-Output "    Min password length       : $minLength (required: >= 14)"
Write-Output "    Password complexity       : $complexity (required: 1)"
Write-Output "    Result                    : $ism0445Pass"
Write-Output ""
Write-Output "  [ISM-1175] Privileged Accounts Blocked from Internet/Network"
Write-Output "    applyPthMitigation        : $applyPthMitigation"
Write-Output "    S-1-5-114 in deny network : $s114Present"
Write-Output "    Detail                    : $ism1175Detail"
Write-Output "    Result                    : $ism1175"
Write-Output "    NOTE: Full ISM-1175 (internet/email block for privileged accounts)"
Write-Output "    is enforced at the SAW and network layer, not on the managed server."
Write-Output ""
Write-Output "  [ISM-1380] Separate Privileged/Unprivileged Environments"
Write-Output "    Result : REQUIRES PROCESS DOCUMENTATION"
Write-Output "    ACTION : Document the use of separate admin workstations or PAWs"
Write-Output "    for all administrative activities. This cannot be verified"
Write-Output "    by a local PowerShell audit on a single server."
Write-Output ""
Write-Output "  [ISM-1507] Privileged Access Requests Validated"
Write-Output "    Result : REQUIRES PROCESS DOCUMENTATION"
Write-Output "    ACTION : Maintain a register of all privileged accounts with"
Write-Output "    documented business justification and approval records."
Write-Output ""
Write-Output "  [ISM-1688] Unprivileged Accounts Cannot Logon to Server"
Write-Output "    Interactive logon restricted to Administrators : $($interactiveLogon -match 'S-1-5-32-544')"
Write-Output "    Result                                         : $ism1688"
Write-Output ""
Write-Output "  [ISM-1689] Privileged Accounts Cannot Logon to Unprivileged Environments"
Write-Output "    Result : REQUIRES PROCESS DOCUMENTATION"
Write-Output "    ACTION : Ensure named admin accounts are not used as standard"
Write-Output "    user accounts on workstations. Enforce via GPO on a domain."
Write-Output ""

$technicalChecks = @($ism0445Pass, $ism1175, $ism1688)

$overallStatus = if ($technicalChecks -contains "FAIL") {
    "NON-COMPLIANT (technical)"
} elseif ($technicalChecks -contains "WARN") {
    "TECHNICALLY COMPLIANT WITH WARNINGS -- review ISM-1175 detail above"
} else {
    "TECHNICALLY COMPLIANT -- process documentation required for ISM-1380, ISM-1507, ISM-1689"
}

Write-Output $separator
Write-Output "  OVERALL STATUS : $overallStatus"
Write-Output $separator

if ($technicalChecks -contains "FAIL") {
    exit 1
}

exit 0
