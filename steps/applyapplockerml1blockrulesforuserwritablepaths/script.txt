$denyPathsExeRaw    = "{denyPathsExe}"
$denyPathsScriptRaw = "{denyPathsScript}"

class AppLockerDenyRuleXml {
    [string]$path

    AppLockerDenyRuleXml([string]$path) {
        $this.path = $path
    }

    [string] ToXml() {
        $ruleId = [guid]::NewGuid().Guid
        $escapedPath = [System.Security.SecurityElement]::Escape($this.path)
        $escapedName = [System.Security.SecurityElement]::Escape("Block Everyone - $($this.path)")
        $escapedDesc = [System.Security.SecurityElement]::Escape("Blocks execution from $($this.path)")

        return @"
    <FilePathRule Id="$ruleId" Name="$escapedName" Description="$escapedDesc" UserOrGroupSid="S-1-1-0" Action="Deny">
      <Conditions>
        <FilePathCondition Path="$escapedPath"/>
      </Conditions>
    </FilePathRule>
"@
    }
}

function Parse-CsvParameter {
    param([string]$raw)

    $result = @()
    foreach ($entry in ($raw -split ",")) {
        $trimmed = $entry.Trim()
        if ($trimmed -eq "") {
            continue
        }
        $result += $trimmed
    }
    return $result
}

function Test-AppLockerPath {
    param([string]$path)

    $supportedVars = @('%WINDIR%', '%SYSTEM32%', '%OSDRIVE%', '%PROGRAMFILES%', '%REMOVABLE%', '%HOT%')

    if ($path -notmatch '^%') {
        return # Literal path - valid, no variable to check
    }

    $firstVar = ($path -split '\\')[0]
    if ($firstVar -notin $supportedVars) {
        Write-Output "WARNING : path '$path' starts with '$firstVar', which is not an AppLocker-recognised variable. Rule will be stored as literal text and will not match real files."
    }
}

$denyPathsExe    = Parse-CsvParameter -raw $denyPathsExeRaw
$denyPathsScript = Parse-CsvParameter -raw $denyPathsScriptRaw

if ($denyPathsExe.Count -eq 0 -and $denyPathsScript.Count -eq 0) {
    Write-Error "Both denyPathsExe and denyPathsScript are empty. At least one deny path is required."
    exit 1
}

Write-Output "Exe deny rules to apply    : $($denyPathsExe.Count)"
Write-Output "Script deny rules to apply : $($denyPathsScript.Count)"

foreach ($path in ($denyPathsExe + $denyPathsScript)) {
    Test-AppLockerPath -path $path
}

$exeRulesXml = @()
foreach ($path in $denyPathsExe) {
    $exeRulesXml += [AppLockerDenyRuleXml]::new($path).ToXml()
}

$scriptRulesXml = @()
foreach ($path in $denyPathsScript) {
    $scriptRulesXml += [AppLockerDenyRuleXml]::new($path).ToXml()
}

$policyXml = @"
<AppLockerPolicy Version="1">
  <RuleCollection Type="Exe" EnforcementMode="AuditOnly">
$($exeRulesXml -join "`n")
  </RuleCollection>
  <RuleCollection Type="Script" EnforcementMode="AuditOnly">
$($scriptRulesXml -join "`n")
  </RuleCollection>
</AppLockerPolicy>
"@

$tempPolicyPath = "$env:TEMP\applocker_deny.xml"
$policyXml | Out-File -FilePath $tempPolicyPath -Encoding UTF8

Set-AppLockerPolicy -XmlPolicy $tempPolicyPath -Merge
Write-Output "Deny rules applied successfully."

Remove-Item -Path $tempPolicyPath -Force
