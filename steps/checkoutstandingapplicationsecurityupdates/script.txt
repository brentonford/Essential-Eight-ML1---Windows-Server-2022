$module = Get-Module -ListAvailable -Name PSWindowsUpdate
if ($null -eq $module) {
    Write-Output "CHECK Outstanding Updates : SKIP — PSWindowsUpdate not available"
    exit 1
}

Import-Module PSWindowsUpdate

$outstanding = Get-WindowsUpdate `
    -Category "Application" `
    -NotTitle "Upgrade to" `
    -ErrorAction SilentlyContinue

if ($null -eq $outstanding -or $outstanding.Count -eq 0) {
    Write-Output "CHECK Outstanding Application Updates : PASS — no updates pending"
    exit 0
}

Write-Output "CHECK Outstanding Application Updates : FAIL — $($outstanding.Count) update(s) pending"
Write-Output ""
Write-Output "Pending application updates:"
$outstanding | ForEach-Object {
    Write-Output "  [$($_.MsrcSeverity)] $($_.Title)"
}
exit 1
