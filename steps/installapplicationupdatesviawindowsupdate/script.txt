Import-Module PSWindowsUpdate

Write-Output "Checking for available application updates via Windows Update..."

$availableUpdates = Get-WindowsUpdate -Category "Application" -NotTitle "Upgrade to" -ErrorAction SilentlyContinue

if ($null -eq $availableUpdates -or $availableUpdates.Count -eq 0) {
    Write-Output "No application updates available via Windows Update."
    exit 0
}

Write-Output "Found $($availableUpdates.Count) application update(s):"
$availableUpdates | ForEach-Object {
    Write-Output "  - $($_.Title) [$($_.MsrcSeverity)]"
}

Write-Output ""
Write-Output "Installing application updates..."

Install-WindowsUpdate `
    -Category "Application" `
    -NotTitle "Upgrade to" `
    -AcceptAll `
    -IgnoreReboot `
    -Confirm:$false `
    -Verbose

Write-Output "Application update installation complete."
